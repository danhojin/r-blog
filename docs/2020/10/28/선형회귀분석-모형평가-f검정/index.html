library(tidyverse)
library(tidymodels)
library(wooldridge)
library(GGally)

data("mlb1")
df <- mlb1
df %>%
  select(salary, years, gamesyr, bavg, hrunsyr, rbisyr) %>%
  ggpairs()

mod.lm <- linear_reg() %>%
  set_engine("lm")
fit.lm.f <- fit(mod.lm,
                log(salary) ~ years + gamesyr + bavg + hrunsyr + rbisyr,
                data = df)

fit.lm.r <- fit(mod.lm,
                log(salary) ~ years + gamesyr,
                data = df)

tidy(fit.lm.f)
