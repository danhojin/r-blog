library(tidyverse)
library(tidymodels)
library(wooldridge)

data("lawsch85")
df <- lawsch85

df %>%
  select(salary, LSAT, GPA, libvol, cost, rank) %>%
  pivot_longer(-salary) %>%
  ggplot(aes(x = value, y = salary, color = name)) +
  geom_point() +
  facet_wrap(~ name, ncol = 2, scales = "free")

library(knitr)
lm_mod <- linear_reg() %>%
  set_engine("lm")
lm_fit <- fit(lm_mod, log(salary) ~ LSAT + GPA + log(libvol) + log(cost) + rank, data = df)
